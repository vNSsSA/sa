    function createZekrElement(text) {
      const div = document.createElement("div");
      div.className = "zekr";

      // نص الذكر
      const p = document.createElement("p");
      p.textContent = text;
      div.appendChild(p);

      // صندوق الأزرار
      const actions = document.createElement("div");
      actions.className = "actions";

      // زر النسخ
      const copyBtn = document.createElement("button");
      copyBtn.textContent = "نسخ";
      copyBtn.title = "نسخ الذكر";
      copyBtn.onclick = () => {
        navigator.clipboard.writeText(text).then(() => {
          alert("تم نسخ الذكر!");
        });
      };
      actions.appendChild(copyBtn);

      // زر القراءة بالصوت
      const speakBtn = document.createElement("button");
      speakBtn.textContent = "🔊";
      speakBtn.title = "تشغيل الصوت";
      speakBtn.onclick = () => {
        if ('speechSynthesis' in window) {
          const utterance = new SpeechSynthesisUtterance(text);
          utterance.lang = "ar-SA";
          speechSynthesis.cancel(); // إيقاف أي قراءة جارية
          speechSynthesis.speak(utterance);
        } else {
          alert("ميزة القراءة الصوتية غير مدعومة في متصفحك.");
        }
      };
      actions.appendChild(speakBtn);

      div.appendChild(actions);

      return div;
    }

    function displayAzkar() {
      const morningContainer = document.getElementById("morning");
      const eveningContainer = document.getElementById("evening");

      azkarMorning.forEach(zekr => {
        morningContainer.appendChild(createZekrElement(zekr));
      });

      azkarEvening.forEach(zekr => {
        eveningContainer.appendChild(createZekrElement(zekr));
      });
    }
